## 1. 引言

### 為什麼需要異步編程？

作為 JavaScript 開發者，你已經熟悉了 Promise、async/await 和事件循環的概念。Python 也有自己的異步編程模型，雖然語法相似，但底層機制有所不同。

**異步編程的核心價值**
- 提高程序性能：避免阻塞 I/O 操作
- 更好的資源利用：單線程處理多個併發任務
- 響應性：保持程序的響應能力
- 可擴展性：支持大量併發連接

> 💡 **學習策略**：將 Python 的異步編程視為 JavaScript 異步模型的"Python 版本"

## 2. 同步 vs 異步

### 2.1 基本概念對比

<PythonEditor title="同步 vs 異步對比" compare={true}>
```javascript !! js
// JavaScript 同步操作
function syncOperation() {
    console.log("開始同步操作");
    // 模擬耗時操作
    const start = Date.now();
    while (Date.now() - start < 2000) {
        // 阻塞 2 秒
    }
    console.log("同步操作完成");
    return "同步結果";
}

// JavaScript 異步操作
async function asyncOperation() {
    console.log("開始異步操作");
    // 模擬異步操作
    await new Promise(resolve => setTimeout(resolve, 2000));
    console.log("異步操作完成");
    return "異步結果";
}

// 同步調用（會阻塞）
console.log("調用同步函數");
const syncResult = syncOperation();
console.log("同步結果:", syncResult);

// 異步調用（不會阻塞）
console.log("調用異步函數");
asyncOperation().then(result => {
    console.log("異步結果:", result);
});
console.log("異步調用後立即執行");
```

## 3. Python 異步編程基礎

### 3.1 async/await 語法

Python 的 async/await 語法與 JavaScript 非常相似，但有一些重要的差異。

<PythonEditor title="async/await 語法對比" compare={true}>
```javascript !! js
// JavaScript async/await
async function fetchUserData(userId) {
    try {
        console.log(`開始獲取用戶 ${userId} 的數據`);
        
        // 模擬 API 調用
        const response = await new Promise((resolve, reject) => {
            setTimeout(() => {
                if (userId > 0) {
                    resolve({
                        id: userId,
                        name: `用戶${userId}`,
                        email: `user${userId}@example.com`
                    });
                } else {
                    reject(new Error("用戶不存在"));
                }
            }, 1000);
        });
        
        console.log("數據獲取成功");
        return response;
        
    } catch (error) {
        console.error("獲取數據失敗:", error.message);
        throw error;
    }
}

async function processUsers() {
    console.log("開始處理用戶數據");
    
    try {
        const user1 = await fetchUserData(1);
        const user2 = await fetchUserData(2);
        
        console.log("用戶1:", user1);
        console.log("用戶2:", user2);
        
        return [user1, user2];
        
    } catch (error) {
        console.error("處理用戶數據失敗:", error.message);
    }
}

// 調用異步函數
processUsers().then(users => {
    console.log("所有用戶:", users);
});
```

### 3.2 併發執行

Python 提供了多種方式來實現併發執行，類似於 JavaScript 的 Promise.all()。

<PythonEditor title="併發執行對比" compare={true}>
```javascript !! js
// JavaScript 併發執行
async function fetchUserData(userId) {
    console.log(`開始獲取用戶 ${userId} 的數據`);
    // 模擬隨機延遲
    const delay = 500 + Math.random() * 1000;
    await new Promise(resolve => setTimeout(resolve, delay));
    console.log(`完成獲取用戶 ${userId} 的數據`);
    return {
        id: userId,
        name: `用戶${userId}`,
        email: `user${userId}@example.com`
    };
}

async function processUsersConcurrently() {
    console.log("開始併發獲取用戶數據 (Promise.all)");
    const promises = [
        fetchUserData(1),
        fetchUserData(2),
        fetchUserData(3)
    ];
    const users = await Promise.all(promises);
    console.log("所有用戶數據:", users);
    return users;
}

// 使用 Promise.race() 獲取最快的結果
async function getFastestUser() {
    console.log("\n開始競速獲取用戶數據 (Promise.race)");
    const promises = [
        fetchUserData(1),
        fetchUserData(2),
        fetchUserData(3)
    ];
    const fastestUser = await Promise.race(promises);
    console.log("最快的用戶:", fastestUser);
    return fastestUser;
}

// 調用函數
processUsersConcurrently();
getFastestUser();
```

```python !! py
import asyncio
import random

# Python 併發執行
async def fetch_user_data(user_id):
    """獲取用戶數據"""
    print(f"開始獲取用戶 {user_id} 的數據")
    # 模擬隨機網絡延遲
    delay = 0.5 + random.random()
    await asyncio.sleep(delay)
    print(f"完成獲取用戶 {user_id} 的數據")
    return {
        "id": user_id,
        "name": f"用戶{user_id}",
        "email": f"user{user_id}@example.com"
    }

async def process_users_concurrently():
    """併發處理用戶數據 (類似 Promise.all)"""
    print("開始併發獲取用戶數據 (asyncio.gather)")
    tasks = [
        fetch_user_data(1),
        fetch_user_data(2),
        fetch_user_data(3)
    ]
    users = await asyncio.gather(*tasks)
    print("所有用戶數據:", users)
    return users

async def get_fastest_user():
    """獲取最快的用戶數據 (類似 Promise.race)"""
    print("\n開始競速獲取用戶數據 (asyncio.wait)")
    tasks = [
        asyncio.create_task(fetch_user_data(1)),
        asyncio.create_task(fetch_user_data(2)),
        asyncio.create_task(fetch_user_data(3))
    ]
    
    # 等待第一個任務完成
    done, pending = await asyncio.wait(tasks, return_when=asyncio.FIRST_COMPLETED)
    
    # 獲取已完成任務的結果
    fastest_user = done.pop().result()
    
    # 取消其他未完成的任務
    for task in pending:
        task.cancel()
        
    print(f"最快的用戶: {fastest_user}")
    return fastest_user

# 運行併發函數
async def main():
    await process_users_concurrently()
    await get_fastest_user()

# 在 Pyodide 環境中，直接調用異步函數
asyncio.create_task(main())
```

### 3.3 異步上下文管理器

Python 的異步上下文管理器類似於 JavaScript 的 try-with-resources 模式。

<PythonEditor title="異步上下文管理器" compare={true}>
```javascript !! js
// JavaScript 資源管理
class AsyncResource {
    constructor(name) {
        this.name = name;
        this.isOpen = false;
    }
    
    async open() {
        console.log(`打開資源: ${this.name}`);
        this.isOpen = true;
    }
    
    async close() {
        console.log(`關閉資源: ${this.name}`);
        this.isOpen = false;
    }
    
    async use() {
        if (!this.isOpen) {
            throw new Error("資源未打開");
        }
        console.log(`使用資源: ${this.name}`);
        await new Promise(resolve => setTimeout(resolve, 1000));
    }
}

// 手動資源管理
async function useResource() {
    const resource = new AsyncResource("數據庫連接");
    
    try {
        await resource.open();
        await resource.use();
    } finally {
        await resource.close();
    }
}

// 使用資源
useResource();
```

```python !! py
import asyncio
from contextlib import asynccontextmanager

# Python 異步上下文管理器
class AsyncResource:
    """異步資源類"""
    
    def __init__(self, name):
        self.name = name
        self.is_open = False
    
    async def __aenter__(self):
        """進入上下文"""
        print(f"打開資源: {self.name}")
        self.is_open = True
        return self
    
    async def __aexit__(self, exc_type, exc_val, exc_tb):
        """退出上下文"""
        print(f"關閉資源: {self.name}")
        self.is_open = False
    
    async def use(self):
        """使用資源"""
        if not self.is_open:
            raise RuntimeError("資源未打開")
        print(f"使用資源: {self.name}")
        await asyncio.sleep(1)

# 使用異步上下文管理器
async def use_resource():
    """使用異步資源"""
    async with AsyncResource("數據庫連接") as resource:
        await resource.use()

# 使用裝飾器創建異步上下文管理器
@asynccontextmanager
async def managed_resource(name):
    """管理的資源上下文管理器"""
    print(f"打開資源: {name}")
    try:
        yield name
    finally:
        print(f"關閉資源: {name}")

async def use_managed_resource():
    """使用管理的資源"""
    async with managed_resource("文件連接") as resource:
        print(f"使用資源: {resource}")
        await asyncio.sleep(1)

# 運行示例
async def main():
    await use_resource()
    await use_managed_resource()

# 在 Pyodide 環境中，直接調用異步函數
asyncio.create_task(main())
```

### 3.4 常見陷阱：異步函數中的阻塞代碼

在異步函數中，絕對不能調用阻塞（同步）的 I/O 操作，否則會阻塞整個事件循環，導致所有併發任務停滯。

**錯誤示例**
```python
import time
import requests  # 同步 HTTP 庫

async def bad_async_function():
    # ❌ 錯誤：這會阻塞整個事件循環！
    time.sleep(5)
    # ❌ 錯誤：這也會阻塞！
    requests.get("https://example.com")
```

**正確做法**
```python
import asyncio
import httpx  # 異步 HTTP 庫

async def good_async_function():
    # ✅ 正確：非阻塞等待
    await asyncio.sleep(5)
    
    # ✅ 正確：使用異步庫
    async with httpx.AsyncClient() as client:
        await client.get("https://example.com")
```

### 3.5 運行異步程序

在不同的環境中，運行異步代碼的方式略有不同。

**1. 標準 Python 腳本**
使用 `asyncio.run()` 作為入口點。

```python
import asyncio

async def main():
    print("Hello")
    await asyncio.sleep(1)
    print("World")

if __name__ == "__main__":
    # 自動創建並管理事件循環
    asyncio.run(main())
```

**2. Jupyter Notebook / Pyodide (本教程環境)**
這些環境通常已經有一個正在運行的事件循環，因此不能使用 `asyncio.run()`。
- 直接 `await async_function()` (如果在支持頂層 await 的環境中)
- 或者 `asyncio.create_task(async_function())` (後臺運行)

```python !! py
# Python 異步數據庫操作
import asyncpg
from typing import List, Optional
from pydantic import BaseModel

class User(BaseModel):
    id: Optional[int] = None
    name: str
    email: str

class DatabaseService:
    """數據庫服務類"""
    
    def __init__(self):
        self.pool = None
    
    async def connect(self):
        """連接數據庫"""
        self.pool = await asyncpg.create_pool(
            host='localhost',
            database='testdb',
            user='username',
            password='password',
            port=5432
        )
    
    async def close(self):
        """關閉數據庫連接"""
        if self.pool:
            await self.pool.close()
    
    async def get_users(self) -> List[User]:
        """獲取所有用戶"""
        async with self.pool.acquire() as conn:
            rows = await conn.fetch('SELECT * FROM users')
            return [User(**dict(row)) for row in rows]
    
    async def get_user_by_id(self, user_id: int) -> Optional[User]:
        """根據 ID 獲取用戶"""
        async with self.pool.acquire() as conn:
            row = await conn.fetchrow(
                'SELECT * FROM users WHERE id = $1',
                user_id
            )
            return User(**dict(row)) if row else None
    
    async def create_user(self, user: User) -> User:
        """創建用戶"""
        async with self.pool.acquire() as conn:
            row = await conn.fetchrow(
                'INSERT INTO users (name, email) VALUES ($1, $2) RETURNING *',
                user.name, user.email
            )
```

```javascript !! js
// JavaScript 異步任務隊列
class AsyncTaskQueue {
    constructor(concurrency = 3) {
        this.concurrency = concurrency;
        this.running = 0;
        this.queue = [];
    }
    
    async add(task) {
        return new Promise((resolve, reject) => {
            this.queue.push({
                task,
                resolve,
                reject
            });
            this.process();
        });
    }
    
    async process() {
        if (this.running >= this.concurrency || this.queue.length === 0) {
            return;
        }
        
        this.running++;
        const { task, resolve, reject } = this.queue.shift();
        
        try {
            const result = await task();
            resolve(result);
        } catch (error) {
            reject(error);
        } finally {
            this.running--;
            this.process();
        }
    }
    
    async waitForAll() {
        while (this.queue.length > 0 || this.running > 0) {
            await new Promise(resolve => setTimeout(resolve, 100));
        }
    }
}

// 使用任務隊列
async function processTask(id) {
    console.log(`開始處理任務 ${id}`);
    await new Promise(resolve => setTimeout(resolve, 1000 + Math.random() * 2000));
    console.log(`完成處理任務 ${id}`);
    return `任務 ${id} 的結果`;
}

async function main() {
    const queue = new AsyncTaskQueue(2);
    
    // 添加任務
    const promises = [];
    for (let i = 1; i <= 5; i++) {
        promises.push(queue.add(() => processTask(i)));
    }
    
    // 等待所有任務完成
    await queue.waitForAll();
    const results = await Promise.all(promises);
    
    console.log('所有任務完成:', results);
}
```

```python !! py
import asyncio
import random
from typing import List, Callable, Any
from dataclasses import dataclass

@dataclass
class Task:
    """任務類"""
    func: Callable
    args: tuple = ()
    kwargs: dict = None
    future: asyncio.Future = None
    
    def __post_init__(self):
        if self.kwargs is None:
            self.kwargs = {}
        if self.future is None:
            self.future = asyncio.Future()

class AsyncTaskQueue:
    """異步任務隊列"""
    
    def __init__(self, concurrency: int = 3):
        self.concurrency = concurrency
        self.running = 0
        self.queue: List[Task] = []
        self.semaphore = asyncio.Semaphore(concurrency)
    
    async def add(self, func: Callable, *args, **kwargs) -> asyncio.Future:
        """添加任務到隊列"""
        task = Task(func, args, kwargs)
        self.queue.append(task)
        asyncio.create_task(self._process_queue())
        return task.future
    
    async def _process_queue(self):
        """處理隊列中的任務"""
        if not self.queue or self.running >= self.concurrency:
            return
        
        self.running += 1
        task = self.queue.pop(0)
        
        try:
            async with self.semaphore:
                result = await task.func(*task.args, **task.kwargs)
                task.future.set_result(result)
        except Exception as error:
            task.future.set_exception(error)
        finally:
            self.running -= 1
            if self.queue:
                asyncio.create_task(self._process_queue())
    
    async def wait_for_all(self):
        """等待所有任務完成"""
        while self.queue or self.running > 0:
            await asyncio.sleep(0.1)

# 使用任務隊列
async def process_task(task_id: int) -> str:
    """處理任務"""
    print(f"開始處理任務 {task_id}")
    await asyncio.sleep(1 + random.random() * 2)
    print(f"完成處理任務 {task_id}")
    return f"任務 {task_id} 的結果"

async def main():
    queue = AsyncTaskQueue(2)
    
    # 添加任務
    futures = []
    for i in range(1, 6):
        future = await queue.add(process_task, i)
        futures.append(future)
    
    # 等待所有任務完成
    await queue.wait_for_all()
    results = await asyncio.gather(*futures)
    
    print("所有任務完成:", results)
```
