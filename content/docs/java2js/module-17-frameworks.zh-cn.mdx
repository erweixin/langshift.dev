## React 基础

<UniversalEditor title="React 组件">
```javascript !! js
// 函数组件
function Welcome(props) {
  return <h1>你好, {props.name}</h1>;
}

// 使用 JSX
function App() {
  const [count, setCount] = useState(0);

  return (
    <div>
      <h1>计数: {count}</h1>
      <button onClick={() => setCount(count + 1)}>
        增加
      </button>
    </div>
  );
}

// Hooks
import { useState, useEffect } from "react";

function UserProfile({ userId }) {
  const [user, setUser] = useState(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    fetchUser(userId).then(data => {
      setUser(data);
      setLoading(false);
    });
  }, [userId]);

  if (loading) return <div>加载中...</div>;
  return <div>{user.name}</div>;
}
```
</UniversalEditor>

## Props 和 State

function Greeting({ name, age }) {
  return (

    </div>

// 使用
<Greeting name="John" age={25} />

// State（可变）
function Counter() {
  const [count, setCount] = useState(0);

  return (
    <div>
      <p>计数: {count}</p>
      <button onClick={() => setCount(count + 1)}>

      </button>

## Vue 基础

<UniversalEditor title="Vue 组件">
```javascript !! js
// Vue 组件
<template>
  <div>
    <h1>{{ message }}</h1>
    <button @click="increment">{{ count }}</button>
  </div>
</template>

<script>
export default {
  data() {
    return {
      message: "你好 Vue！",
      count: 0
    };
  },
  methods: {
    increment() {
      this.count++;
    }
  }
};
</script>
```

## 总结

### 关键要点

1. **React**：基于组件、hooks、JSX
2. **Vue**：基于模板、响应式
3. **状态管理**：useState、Vuex
4. **组件**：可重用的 UI 片段
5. **生态系统**：丰富的库

### 对比：Java vs JavaScript

| 特性 | Java | JavaScript |
|---------|------|------------|
| **UI** | Swing/JavaFX | React/Vue |
| **状态** | 字段 | useState |
| **事件** | 监听器 | onClick/@click |
| **组件** | 类 | 函数 |
